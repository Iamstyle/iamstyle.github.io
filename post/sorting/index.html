四种排序算法：

冒泡，选择，插入，快速

源码如下：

#include <iostream>
using namespace std;

//冒泡排序
void bubbleSort(int arr[], int n) {
	//从最大值处遍历数组，每次遍历完最后一个就为最大，然后j--将最大的忽略
	for (int j = n; j > 1; j--) {
		//每次遍历前n-1个，数组相邻的数顺序错误就两两交换（最后一个为最大）
		for (int i = 0; i < j - 1; i++) {
			if (arr[i] > arr[i + 1])
			{
				//两两交换
				int temp = arr[i];
				arr[i] = arr[i + 1];
				arr[i + 1] = temp;
			}
		}
	}

}

//选择排序
void selectionSort(int arr[], int n) {
	//只剩下坐标为0一个数就是最小，只需要比较到下坐标为1然后插入到下坐标为1的位置即可
	for (; n >= 1;n--) {
		int pos = 0; //记录当前数组中最大值的位置
		int max = arr[0];
		//遍历当前数组找出最大的数值的位置
		for (int i = 0; i < n; i++) {
			if (max < arr[i])
			{
				max = arr[i];
				pos = i;
			}
		}
		//如果pos不在当前数组的最大位处就交换数值
		if (pos != n - 1) {
			int temp = arr[n - 1];
			arr[n - 1] = arr[pos];
			arr[pos] = temp;
		}
	}
}

//插入排序
void insertionSort(int arr[], int n) {
	//从下标为1开始向前插入
	for (int i = 1; i < n; i++) {
		int key = arr[i];//key为插入数的值
		int j = i;
		/*从插入数组最后一位开始向前遍历，如果大于插入数则将数后移一位
		直到其小于等于数组中的一个数或成为数组中最小的数*/
		for (; arr[j-1] > key; j--) {
			arr[j] = arr[j - 1];
			//插入数比数组中所有数小就跳出
			if (j == 0)
			{
				break;
			}
		}
		//插入数的位置为j
		arr[j] = key;
	}
}

//快速排序
void quickSort(int arr[], int left, int right) {
	if (left >= right)//递归边界条件
		return;
	int pivod = arr[left];//记录基准数
	//新定义
	int l, r;
	l = left;
	r = right;
	/*
		问题：由于数组特殊，以左为基准只能先向左寻找即r--，如果先向右寻找即l++得到排序后的数组错误
	*/
	//循环寻找直到左右标记数相遇
	while (l < r) {
		//向左寻找比基准数小的数
		while (arr[r] >= pivod && l < r) {
			r--;
		}
		//向右寻找比基准数大的数
		while (arr[l] <= pivod && l < r) {
			l++;
		}
		//数找到后进行一次交换
		if (l < r) {
			int temp = arr[l];
			arr[l] = arr[r];
			arr[r] = temp;
		}
	}
	//基准数归位
	arr[left] = arr[l];
	arr[l] = pivod;
	//递归进行寻找
	quickSort(arr, left, l - 1);
	quickSort(arr, l + 1, right);
}
int main()
{
	int arr[] = {2,5,9,8,6,3,1,7,4}; //2 1 9 8 6 3 5 7 4
	int n = sizeof(arr) / sizeof(arr[0]);//计算数组长度

	//bubbleSort(arr, n);
	//selectionSort(arr, n);
	//insertionSort(arr, n);
	quickSort(arr, 0, n - 1);


	//打印排序后的数组
	for (int i = 0; i < n; i++) {
		cout << arr[i] << " ";
	}
	return 0;
}
